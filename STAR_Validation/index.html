sql = """SELECT
  sample_tag.*,
  series_tag.regex,
  first_name,
  last_name,
  tag_name,
  date_trunc('day', sample_tag.created_on) as trunc_created_on
FROM sample_tag
  INNER JOIN series_tag ON sample_tag.series_tag_id = series_tag.id
  INNER JOIN tag ON tag_id = tag.id
  INNER JOIN auth_user ON sample_tag.created_by = auth_user.id
ORDER BY sample_tag.id"""

sample_tag = pd.read_sql_query(sql, conn)\
        .sort('trunc_created_on') \
        .set_index('trunc_created_on')

meta['fixed_signed'] = meta.pval_fixed * meta.apply(lambda x: -1 if x['te_fixed'] < 0 else 1, axis=1)
meta['random_signed'] = meta.pval_random * meta.apply(lambda x: -1 if x['te_random'] < 0 else 1, axis=1)

# perform clustering and plot the dendrogram
from scipy.cluster.hierarchy import linkage, dendrogram
df = meta.groupby(['mygene_entrez']) \
        .filter(lambda x: x.analysis_id.count() == len(names.index)) \

        .fixed_signed.unstack()
